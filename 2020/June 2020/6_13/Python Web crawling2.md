- 사이트 마다 태그명이 다르기 떄문에 내가 어떤 데이터를 가져올 것인가에 대해 태그명과 클래스명을 잘 정해서 가져오는 것이 좋다 

#### 알라딘 사이트 크롤링

```python
from selenium import webdriver
from bs4 import BeautifulSoup
import time
import codecs


file = codecs.open('C:/Users/82105/온라인강의 자료/Aladin Best Seller.txt', encoding="utf-8", mode="w")
driver = webdriver.Chrome('C:/Users/82105/온라인강의 자료/chromedriver.exe')
driver.get("https://www.aladin.co.kr/shop/common/wbest.aspx?BranchType=1&start=we")

source = driver.page_source

total_data = BeautifulSoup(source, "html.parser")

ss_book_box = total_data.find_all("div", class_="ss_book_box")

# 가져온 자료가 리스트인지 뷰티플스프자료인지 확인해야함
# li 태그만 모아논 리스트 추출
# 제목 저자 가격 1
file.write("-- Aladin Best Seller --\r\n")
index = 1
for item in ss_book_box:
    li = item.find_all("li")
    file.write(str(index) + "위 도서목록\r\n")
    if len(li) < 8:
        file.write(li[0].text + "\r\n")
        file.write(li[1].text + "\r\n")
        file.write(li[2].text + "\r\n")
    else:
        file.write(li[0].text + "\r\n")
        file.write(li[1].text + "\r\n")
        file.write(li[2].text + "\r\n")
    file.write("-"*50 + "\r\n")
    index += 1


time.sleep(3)
driver.close()
file.close()

```
